% Frame title template (see beamer user guide, p. 69)
\setbeamertemplate{frametitle}{%
  % Undo space reservation by (partial) headline
  \vspace*{-13.5mm}%
  % Disable caching
  \tikzset{external/export next=false}%
  \begin{tikzpicture}
    % Picture size
    \useasboundingbox (0mm, {\headlineHeight - 6mm}) rectangle ({\paperwidth - \leftSidebarWidth- \leftTextMargin - \rightSidebarWidth - \rightTextMargin}, 0mm);

    % Frame title
    \node [anchor=north west, no sep] at (0mm, {\headlineHeight - 6mm - 3.75mm}) {%
      \usebeamerfont{frametitle}%
      \usebeamercolor[fg]{frametitle}%
      \insertframetitle%
    };

    % Frame subtitle
    \node [anchor=north west, no sep] at (0mm, {\headlineHeight - 6mm - 10mm}) {%
      \usebeamerfont{framesubtitle}%
      \usebeamercolor[fg]{framesubtitle}%
      \insertframesubtitle%
    };
  \end{tikzpicture}%
}
