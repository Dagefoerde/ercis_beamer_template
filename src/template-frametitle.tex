% Frame title template (see beamer user guide, p. 69)
\makeatletter
\setbeamertemplate{frametitle}{%
  % Undo space reservation by (partial) headline
  \vspace*{-13.5mm}%
  % Disable caching
  \tikzset{external/export next=false}%
  \begin{tikzpicture}
    % Picture size
    \useasboundingbox (0mm, {\headlineHeight - 6mm}) rectangle ({\paperwidth - \leftSidebarWidth- \leftTextMargin - \rightSidebarWidth - \rightTextMargin}, 0mm);

    % Frame title
    \ifbeamer@renderheadlinetitle
      \node [anchor=text, no sep, align=flush left, text width={\paperwidth - \leftSidebarWidth- \leftTextMargin - \rightSidebarWidth - \rightTextMargin - 23.2mm - 2mm}] at (0mm, {\headlineHeight - 6mm - 7mm}) {%
        \usebeamerfont{frametitle}%
        \usebeamercolor[fg]{frametitle}%
        \insertframetitle%
      };
    \fi

    % Frame subtitle
    \ifbeamer@renderheadlinesubtitle
      \node [anchor=text, no sep, align=flush left, text width={\paperwidth - \leftSidebarWidth- \leftTextMargin - \rightSidebarWidth - \rightTextMargin - 23.2mm - 2mm}] at (0mm, {\headlineHeight - 6mm - 12mm}) {%
        \usebeamerfont{framesubtitle}%
        \usebeamercolor[fg]{framesubtitle}%
        \insertframesubtitle%
      };
    \fi
  \end{tikzpicture}%
}
\makeatother
