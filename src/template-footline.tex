% Footline template (see beamer user guide, p. 61)
\makeatletter
\setbeamertemplate{footline}{%
  \ifbeamer@renderfootline
    {%
      \tikzset{external/export=false}%
      \begin{tikzpicture}
        % Picture size
        \useasboundingbox (0mm, \footlineHeight) rectangle (\paperwidth, 0mm);

        % Background
        \usebeamercolor{footline}
        \path [fill, color={footline.bg}] (0mm, 11mm) rectangle ({\paperwidth - 5mm}, 4.25mm);

        % Title
        \node[anchor=west, no sep] at (6mm, {11mm - 2mm}) {%
          \usebeamerfont{title in foot}%
          \usebeamercolor[fg]{title in foot}%
          \insertshorttitle%
        };

        % Author
        \node[anchor=west, no sep] at (6mm, {11mm - 2mm - 2.5mm}) {%
          \usebeamerfont{author in foot}%
          \usebeamercolor[fg]{author in foot}%
          \insertauthor%
        };

        % Frame number
        \node[anchor=east, no sep] at ({\paperwidth - 5mm - 1.75mm}, {11mm - 2mm}) {%
          \usebeamerfont{frame number in foot}%
          \usebeamercolor[fg]{frame number in foot}%
          \insertframenumber%
        };

        % Date
        \node[anchor=east, no sep] at ({\paperwidth - 5mm - 1.75mm}, {11mm - 2mm - 2.5mm}) {%
          \usebeamerfont{date in foot}%
          \usebeamercolor[fg]{date in foot}%
          \insertdate%
        };

        % Navigation symbols (optional)
        % \ifbeamertemplateempty{navigation symbols}{}{%
        %   \node[anchor=west, no sep] at (7.5mm, 2mm) {%
        %     \usebeamertemplate***{navigation symbols}%
        %   };
        % }
      \end{tikzpicture}%
    }%
  \fi%
}
\makeatother
